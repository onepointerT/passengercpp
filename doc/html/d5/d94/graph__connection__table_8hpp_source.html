552 tree_t& make_tree_from_to( const NodeT* node_from, const NodeT* node_to ) {
553
554 OutgoingMap& outgoings_start = this->make_outgoingmap( node_from );
555 typename OutgoingMap::iterator nodes_start_it = outgoings_start.begin();
556 NodeT* node_current = nodes_start_it->second;
557
558 tree_t* tree = new tree_t( node_from );
559 if ( node_current == node_to ) return *tree;
560
561 IncomingMap& incoming = this->make_incomingmap( node_to );
562
563 while ( node_current != node_to ) {
564 bool has_leaf_to_node_to = false;
565
566 for ( typename IncomingMap::iterator in_end_it = incoming.begin()
567 ; in_end_it != incoming.end(); ++in_end_it
568 ) {
569 bool is_route = false;
570 node_route_t* node_route = Route::findRouteToFrom( in_end_it, node_current );
571
572 if ( (*node_route)[node_route->size()-1] == node_current ) is_route = true;
573
574 if ( ! is_route ) {
575 tree_t& tree_new = this->make_tree_from_to( node_current, in_end_it->second );
576
577 tree.add( node_current, tree_new );
578 } else {
579 typename tree_t::node_t* current_tree_node = tree.find( (*node_route)[0] );
580
581 if ( current_tree_node == nullptr ) current_tree_node = tree.root;
582
583 for ( NodeT* node : *node_route ) {
584
585 typename tree_t::node_t* tree_node_new
586 = new tree_t::node_t( node );
587 tree.add( current_tree_node, tree_node_new );
588
589 if ( node == node_to ) {
590 typename tree_t::leaf_t* new_leaf
591 = new tree_t::leaf_t( node_to );
592 tree.add( new_leaf, current_tree_node );
593 }
594
595 current_tree_node = tree_node_new;
596 }
597 }
598
599 if ( ! has_leaf_to_node_to ) continue;
600
601 if ( in_end_it == incoming.node_end ) break;
602 }
603
604 ++nodes_start_it;
605
606 if ( nodes_start_it == incoming.end() ) node_current = node_to;
607 else node_current = nodes_start_it->second;
608 }
609
610 return *tree;
611 }

664 node_route_t* make_route_along_nodes( const NodeT* start_node, const NodeT* end_node, bool shortest = false ) {
665 node_route_t* node_route = new node_route_t();
666
667 tree_t& tree;
668 if ( shortest )
669 tree_t& tree = this->make_complete_tree_from_to( start_node, end_node );
670 else
671 tree_t& tree = this->make_tree_from_to( start_node, end_node );
672
673 // Select the shortest path
674 unsigned int shortest_il = 0;
675 unsigned int shortest_v = 0;
676 unsigned int il = 0;
677 typename tree_t::leaf_t& leaf;
678 for ( typename tree_t::leaf_t& leaf : tree.leafs ) {
679 const unsigned int depth_to_leaf = tree.depthTo( leaf );
680 if ( shortest_v == 0 || depth_to_leaf < shortest_il ) {
681 shortest_il = il;
682 shortest_v = depth_to_leaf;
683 }
684
685 il++;
686 }
687
688 typename tree_t::node_t* node = leaf.parent;
689 node_route.push_back( leaf.content );
690 while ( node != tree.root ) {
691 node_route.push_back( node );
692 node = node.parent;
693 }
694
695 node_route.push_back( tree.root );
696
697 return node_route;
698 }
